def plot_cat(col, df):
    '''Takes the column name and dataframe, returns the counts and percentage'''
    pct_cnt = pd.DataFrame(df.groupby(col)[col].count())
    pct_cnt['PercentStatus'] = (pct_cnt[col]/pct_cnt[col].sum())*100
    pct_cnt.sort_values(by='PercentStatus', ascending=False)
    
    return pct_cnt
